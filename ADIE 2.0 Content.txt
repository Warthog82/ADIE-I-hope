def require_admin(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        auth_header = request.headers.get("Authorization")
        if not auth_header:
            return jsonify({"error": "Authorization header missing"}), 401
        try:
            token = auth_header.split("Bearer ")[1]
            decoded_token = auth.verify_id_token(token)
            if decoded_token.get("role") != "admin":
                return jsonify({"error": "Admin access required"}), 403
            request.uid = decoded_token["uid"]
        except Exception as e:
            return jsonify({"error": f"Invalid token: {str(e)}"}), 403
        return f(*args, **kwargs)
    return decorated_function

def encrypt_data(data, key_name):
    plaintext = json.dumps(data).encode("utf-8")
    response = kms_client.encrypt(request={"name": key_name, "plaintext": plaintext})
    return response.ciphertext

def decrypt_data(ciphertext, key_name):
    response = kms_client.decrypt(request={"name": key_name, "ciphertext": ciphertext})
    return json.loads(response.plaintext.decode("utf-8"))
